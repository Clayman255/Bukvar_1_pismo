<script>
  const totalPairs = 10; // 10 пар = 20 карток
  let cardsData = [];

  function generateCardsData() {
    let temp = [];
    // Використовуємо зображення від Img_0.png до Img_9.png
    for (let i = 0; i < totalPairs; i++) {
      // Додаємо дві однакові картки для пари
      temp.push({
        id: i, // Унікальний ID пари
        img: `Img_${i}.png`
      });
      temp.push({
        id: i,
        img: `Img_${i}.png`
      });
    }
    return temp;
  }

  let cards = []; 
  let flippedCards = []; 
  let lockBoard = false;
  let matchedCount = 0;

  const gameBoard = document.getElementById('gameBoard');
  const successMessage = document.getElementById('successMessage');
  const restartBtn = document.getElementById('restartBtn');

  function createCards() {
    cards = generateCardsData();
    cards.sort(() => 0.5 - Math.random()); // Перемішуємо
    
    gameBoard.innerHTML = '';
    matchedCount = 0;
    successMessage.classList.remove('visible');
    restartBtn.style.display = 'none';

    cards.forEach((item) => {
      const card = document.createElement('div');
      card.classList.add('card');
      card.dataset.pairId = item.id; 

      card.innerHTML = `
        <div class="card-inner">
          <div class="card-front"></div>
          <div class="card-back">
            <img src="${item.img}" alt="Казковий герой">
          </div>
        </div>
      `;

      card.addEventListener('click', () => handleCardClick(card));
      gameBoard.appendChild(card);
    });
  }

  function handleCardClick(card) {
    if (lockBoard || card.classList.contains('flipped') || card.classList.contains('matched')) return;

    card.classList.add('flipped');
    flippedCards.push(card);

    if (flippedCards.length === 2) {
      lockBoard = true;
      const [card1, card2] = flippedCards;
      
      const isMatch = card1.dataset.pairId === card2.dataset.pairId;

      if (isMatch) {
        setTimeout(() => {
          card1.classList.add('matched');
          card2.classList.add('matched');
        }, 300);

        setTimeout(() => {
          card1.classList.add('hidden');
          card2.classList.add('hidden');
          
          resetBoard();
          matchedCount += 2;
          
          if (matchedCount === cards.length) {
            showSuccess();
          }
        }, 1500); 

      } else {
        setTimeout(() => {
          card1.classList.remove('flipped');
          card2.classList.remove('flipped');
          resetBoard();
        }, 1000);
      }
    }
  }

  function resetBoard() {
    flippedCards = [];
    lockBoard = false;
  }

  function showSuccess() {
    successMessage.classList.add('visible');
    restartBtn.style.display = 'inline-block';
  }

  restartBtn.addEventListener('click', createCards);

  // Запуск гри
  createCards();
</script>
